#include "dirent.h"
#include "iostream"

int main()
{

	struct dirent *entry = nullptr;
	DIR *dp = nullptr;
	string filepath = "";
	dp = opendir(filepath.c_str());
	if (dp != nullptr)
	{
		while ((entry = readdir(dp)))
		{
			string fileName = entry->d_name;
			cout << fileName << endl;
		}
	}
	closedir(dp);

	return 0;

}
